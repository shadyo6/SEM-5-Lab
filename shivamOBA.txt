Execute the following commands on a UNIX system.
[root@localhost ~]# cd /tmp/
[root@localhost tmp]# pwd
/tmp
[root@localhost tmp]# ls
[root@localhost tmp]# mkdir folder1 folder2 folder3
[root@localhost tmp]# ls
folder1  folder2  folder3
[root@localhost tmp]# rmdir folder2
[root@localhost tmp]# ls
folder1  folder3
[root@localhost tmp]# mv folder3 shivam
[root@localhost tmp]# ls
folder1  shivam
[root@localhost tmp]# echo "This is shivam karmudi, hello ya'all" > newfile.txt
[root@localhost tmp]# ls
folder1  newfile.txt  shivam
[root@localhost tmp]# cat newfile.txt 
This is shivam karmudi, hello ya'all
[root@localhost tmp]# touch newshivamfile
[root@localhost tmp]# ls
folder1  newfile.txt  newshivamfile  shivam
[root@localhost tmp]# cat newshivamfile 
[root@localhost tmp]# cp newfile.txt newshivamfile 
cp: overwrite `newshivamfile'? yes
[root@localhost tmp]# ls
folder1  newfile.txt  newshivamfile  shivam
[root@localhost tmp]# cat newshivamfile 
This is shivam karmudi, hello ya'all
[root@localhost tmp]# rm -rf *
[root@localhost tmp]# ls
